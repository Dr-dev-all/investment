import { NextResponse, NextRequest } from 'next/server';

async function middleware(req: NextRequest) {
  //   console.log("first");

  console.log({ pathname: req.nextUrl.pathname });
  console.log({ origin: req.nextUrl.origin });

  const protectedRotes = ['/login/userdash', '/login/adminDash/'];
  try {
    const response = await fetch('http://127.0.0.1:5000/auths/getusertoken', {
      method: 'GET',
      headers: { 'Content-Type': 'application/json' },
    });

    const { token } = await response.json();
    if (!token && protectedRotes.includes(req.nextUrl.pathname)) {
      const absoluteURL = new URL('/', req.nextUrl.origin);
      NextResponse.redirect(absoluteURL.toString());
    }
  } catch (error) {
    NextResponse.redirect('http://127.0.0..1:3000/login');
  }
}

export default middleware;
